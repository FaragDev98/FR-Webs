// منع كليك يمين window.addEventListener("contextmenu", function (e) { e.preventDefault(); }); // منع اختصارات النسخ، الحفظ، عرض الكود، الأدوات window.addEventListener("keydown", function (e) { const key = e.key.toLowerCase(); if ( (e.ctrlKey && ["c", "u", "s", "p", "a"].includes(key)) || e.key === "F12" || (e.ctrlKey && e.shiftKey && ["i", "j"].includes(key)) ) { e.preventDefault(); alert("🚫 هذا الإجراء غير مسموح."); } }); // منع تحديد النص (يسمح بالأرقام والروابط فقط) document.addEventListener("selectstart", function (e) { const selection = window.getSelection().toString(); const isAllowed = /^[0-9:/.\s\-]+$/.test(selection); if (!isAllowed) { e.preventDefault(); window.getSelection().removeAllRanges(); } }); // منع تصوير الشاشة بزر PrintScreen window.addEventListener("keyup", function (e) { if (e.key === "PrintScreen") { navigator.clipboard.writeText("🚫 لا يمكنك تصوير هذه الصفحة"); alert("📸 تم حظر لقطة الشاشة!"); } });
